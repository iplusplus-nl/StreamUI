<script>
    const grid = document.getElementById('grid');
    const cellsDOM = Array.from(document.querySelectorAll('.cell'));
    const scoreDOM = document.getElementById('score');
    const bestScoreDOM = document.getElementById('best-score');
    const gameOverOverlay = document.getElementById('game-over');

    let board = [];
    let score = 254; // Matches the sum of the mid-game seed values
    let bestScore = 254;

    // Initialize game state exactly from the predefined HTML layout
    function init() {
        board = [];
        for (let r = 0; r < 4; r++) {
            let row = [];
            for (let c = 0; c < 4; c++) {
                let val = cellsDOM[r * 4 + c].getAttribute('data-val');
                row.push(val ? parseInt(val) : 0);
            }
            board.push(row);
        }
    }

    function resetGame() {
        board = Array(4).fill().map(() => Array(4).fill(0));
        score = 0;
        gameOverOverlay.classList.add('hidden');
        spawnTile();
        spawnTile();
        updateDOM();
    }

    function spawnTile() {
        let emptySpots = [];
        for (let r = 0; r < 4; r++) {
            for (let c = 0; c < 4; c++) {
                if (board[r][c] === 0) emptySpots.push({ r, c });
            }
        }
        if (emptySpots.length > 0) {
            let spot = emptySpots[Math.floor(Math.random() * emptySpots.length)];
            board[spot.r][spot.c] = Math.random() < 0.9 ? 2 : 4;
        }
    }

    function checkGameOver() {
        for (let r = 0; r < 4; r++) {
            for (let c = 0; c < 4; c++) {
                if (board[r][c] === 0) return false;
                if (c < 3 && board[r][c] === board[r][c + 1]) return false;
                if (r < 3 && board[r][c] === board[r + 1][c]) return false;
            }
        }
        gameOverOverlay.classList.remove('hidden');
        return true;
    }

    function updateDOM() {
        for (let r = 0; r < 4; r++) {
            for (let c = 0; c < 4; c++) {
                let val = board[r][c];
                let cell = cellsDOM[r * 4 + c];
                let oldVal = cell.getAttribute('data-val');

                cell.setAttribute('data-val', val > 0 ? val : "");
                cell.textContent = val > 0 ? val : "";

                // Pop animation for newly spawned or merged tiles
                if (val > 0 && String(val) !== oldVal && oldVal !== null) {
                    cell.classList.add('pop');
                    setTimeout(() => cell.classList.remove('pop'), 150);
                }
            }
        }
        scoreDOM.innerText = score;
        if (score > bestScore) {
            bestScore = score;
            bestScoreDOM.innerText = bestScore;
        }
    }

    // Core movement logic functions target single row vectors
    function slide(row) {
        let arr = row.filter(val => val);
        let zeros = Array(4 - arr.length).fill(0);
        return arr.concat(zeros);
    }

    function combine(row) {
        for (let i = 0; i < 3; i++) {
            if (row[i] !== 0 && row[i] === row[i + 1]) {
                row[i] *= 2;
                score += row[i];
                row[i + 1] = 0;
            }
        }
        return row;
    }

    function operate(row) {
        return slide(combine(slide(row)));
    }

    // Directional actions
    function moveLeft() {
        let moved = false;
        for (let i = 0; i < 4; i++) {
            let row = board[i];
            let newRow = operate(row);
            if (row.join(',') !== newRow.join(',')) moved = true;
            board[i] = newRow;
        }
        return moved;
    }

    function moveRight() {
        let moved = false;
        for (let i = 0; i < 4; i++) {
            let row = board[i].slice().reverse();
            let newRow = operate(row).reverse();
            if (board[i].join(',') !== newRow.join(',')) moved = true;
            board[i] = newRow;
        }
        return moved;
    }

    function moveUp() {
        let moved = false;
        for (let i = 0; i < 4; i++) {
            let col = [board[0][i], board[1][i], board[2][i], board[3][i]];
            let newCol = operate(col);
            for (let j = 0; j < 4; j++) {
                if (board[j][i] !== newCol[j]) moved = true;
                board[j][i] = newCol[j];
            }
        }
        return moved;
    }

    function moveDown() {
        let moved = false;
        for (let i = 0; i < 4; i++) {
            let col = [board[0][i], board[1][i], board[2][i], board[3][i]].reverse();
            let newCol = operate(col).reverse();
            for (let j = 0; j < 4; j++) {
                if (board[j][i] !== newCol[j]) moved = true;
                board[j][i] = newCol[j];
            }
        }
        return moved;
    }

    function attemptMove(moveFunc) {
        if (!gameOverOverlay.classList.contains('hidden')) return;
        let moved = moveFunc();
        if (moved) {
            spawnTile();
            updateDOM();
            checkGameOver();
        }
    }

    // Input Listeners
    document.addEventListener('keydown', e => {
        if (["ArrowUp", "ArrowDown", "ArrowLeft", "ArrowRight"].includes(e.key)) {
            e.preventDefault();
            if (e.key === "ArrowLeft") attemptMove(moveLeft);
            if (e.key === "ArrowRight") attemptMove(moveRight);
            if (e.key === "ArrowUp") attemptMove(moveUp);
            if (e.key === "ArrowDown") attemptMove(moveDown);
        }
    });

    let touchStartX = 0, touchStartY = 0;
    grid.addEventListener('touchstart', e => {
        touchStartX = e.changedTouches[0].screenX;
        touchStartY = e.changedTouches[0].screenY;
    }, {passive: true});

    grid.addEventListener('touchend', e => {
        let dx = e.changedTouches[0].screenX - touchStartX;
        let dy = e.changedTouches[0].screenY - touchStartY;

        if (Math.abs(dx) > Math.abs(dy)) {
            if (Math.abs(dx) > 30) {
                if (dx > 0) attemptMove(moveRight);
                else attemptMove(moveLeft);
            }
        } else {
            if (Math.abs(dy) > 30) {
                if (dy > 0) attemptMove(moveDown);
                else attemptMove(moveUp);
            }
        }
    }, {passive: true});

    document.getElementById('new-game').addEventListener('click', resetGame);
    document.getElementById('try-again').addEventListener('click', resetGame);

    // Boot up
    init();
    updateDOM();
</script>
